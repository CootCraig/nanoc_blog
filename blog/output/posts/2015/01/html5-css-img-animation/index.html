           if(e.keyCode == 8){
               // user has pressed backspace
               array.pop();
           }
           if(e.keyCode == 32){
               // user has pressed space
               array.push('');
           }
        });

          $('#inputid').bind('keypress', function(e) {
            //space bar

              e.preventDefault();
              alert('space');
            }
            //backspace
            if (e.which == 8) {
              e.preventDefault();
              alert('back space');
            }
          });
        });
